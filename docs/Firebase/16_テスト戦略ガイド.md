## 1. Firebase アプリのテスト戦略概要

```
テストピラミッド

        ┌──────────┐
        │  E2E テスト  │  ← 少数・高コスト（Playwright）
        │  （実際のUI） │    本番に近い環境で重要フローのみ
        └────┬─────┘
        ┌────┴──────────┐
        │  統合テスト      │  ← 中程度（Jest + Firebase Emulator）
        │  （エミュレーター）│    Firebase との連携を実際に検証
        └────┬───────────┘
   ┌─────────┴────────────────┐
   │     単体テスト             │  ← 多数・低コスト（Jest + Mock）
   │  （ロジック・コンポーネント）│    Firebase SDK はモックに差し替え
   └──────────────────────────┘
```

| テスト種類 | ツール | Firebase の扱い | 速度 | カバー範囲 |
|----------|-------|--------------|------|----------|
| 単体テスト | Vitest / Jest | SDK をモック | ⚡ 速い | ビジネスロジック・ユーティリティ |
| 統合テスト | Jest + Emulator | エミュレーターに実接続 | 🔶 中程度 | CRUD・セキュリティルール |
| E2E テスト | Playwright | エミュレーター or 本番 | 🐢 遅い | ユーザーシナリオ全体 |

### ユーティリティ関数のテスト

```typescript
// src/lib/firebase/auth.ts（テスト対象）
import { auth } from './client';
import { signInWithEmailAndPassword, createUserWithEmailAndPassword } from 'firebase/auth';

export async function signIn(email: string, password: string) {
  return signInWithEmailAndPassword(auth, email, password);
}

export function getFirebaseErrorMessage(code: string): string {
  const messages: Record<string, string> = {
    'auth/email-already-in-use': 'このメールアドレスはすでに使用されています',
    'auth/wrong-password':       'パスワードが間違っています',
    'auth/user-not-found':       'ユーザーが見つかりません',
    'auth/invalid-email':        'メールアドレスの形式が正しくありません',
  };
  return messages[code] ?? '予期せぬエラーが発生しました';
}
```

```typescript
// tests/unit/auth.test.ts
import { describe, it, expect, vi, beforeEach } from 'vitest';
import { signIn, getFirebaseErrorMessage } from '@/lib/firebase/auth';
import * as firebaseAuth from 'firebase/auth';

// firebase/auth モジュール全体をモック
vi.mock('firebase/auth', async (importOriginal) => {
  const actual = await importOriginal<typeof firebaseAuth>();
  return {
    ...actual,
    signInWithEmailAndPassword: vi.fn(),
  };
});

describe('signIn', () => {
  beforeEach(() => vi.clearAllMocks());

  it('正常にサインインできる', async () => {
    const mockUser = { uid: 'user-123', email: 'test@example.com' };
    vi.mocked(firebaseAuth.signInWithEmailAndPassword).mockResolvedValueOnce({
      user: mockUser,
    } as any);

    const result = await signIn('test@example.com', 'password123');
    expect(result.user.uid).toBe('user-123');
    expect(firebaseAuth.signInWithEmailAndPassword).toHaveBeenCalledWith(
      expect.anything(),
      'test@example.com',
      'password123'
    );
  });

  it('認証エラーをスローする', async () => {
    vi.mocked(firebaseAuth.signInWithEmailAndPassword).mockRejectedValueOnce({
      code: 'auth/wrong-password',
    });

    await expect(signIn('test@example.com', 'wrong')).rejects.toMatchObject({
      code: 'auth/wrong-password',
    });
  });
});

describe('getFirebaseErrorMessage', () => {
  it.each([
    ['auth/email-already-in-use', 'このメールアドレスはすでに使用されています'],
    ['auth/wrong-password',       'パスワードが間違っています'],
    ['auth/user-not-found',       'ユーザーが見つかりません'],
    ['auth/unknown-code',         '予期せぬエラーが発生しました'],
  ])('エラーコード %s を正しく変換する', (code, expected) => {
    expect(getFirebaseErrorMessage(code)).toBe(expected);
  });
});
```

### Firestore ユーティリティのモック

```typescript
// src/lib/firebase/posts.ts（テスト対象）
import { db } from './client';
import { collection, addDoc, getDocs, query, where, orderBy, Timestamp } from 'firebase/firestore';

export type Post = { id: string; title: string; authorId: string; createdAt: Date };

export async function createPost(title: string, authorId: string): Promise<string> {
  if (!title.trim()) throw new Error('タイトルは必須です');
  const docRef = await addDoc(collection(db, 'posts'), {
    title, authorId, createdAt: Timestamp.now(),
  });
  return docRef.id;
}

export async function getMyPosts(authorId: string): Promise<Post[]> {
  const q = query(
    collection(db, 'posts'),
    where('authorId', '==', authorId),
    orderBy('createdAt', 'desc')
  );
  const snapshot = await getDocs(q);
  return snapshot.docs.map((doc) => ({
    id: doc.id,
    ...(doc.data() as Omit<Post, 'id'>),
    createdAt: doc.data().createdAt.toDate(),
  }));
}
```

```typescript
// tests/unit/posts.test.ts

    collection: vi.fn(),

    where:    vi.fn(),
```

## 4. エミュレーターを使った統合テスト

```typescript
// tests/integration/firestore.test.ts
import { describe, it, expect, beforeAll, afterAll, afterEach } from 'vitest';
import {
  initializeTestEnvironment,
  assertSucceeds,
  assertFails,
  RulesTestEnvironment,
} from '@firebase/rules-unit-testing';
import { doc, setDoc, getDoc, deleteDoc, collection, addDoc } from 'firebase/firestore';

let testEnv: RulesTestEnvironment;

beforeAll(async () => {
  testEnv = await initializeTestEnvironment({
    projectId: 'demo-test',
    firestore: { host: '127.0.0.1', port: 8080 },
  });
});

afterEach(async () => {
  await testEnv.clearFirestore(); // テスト間でデータをリセット
});

afterAll(async () => {
  await testEnv.cleanup();
});

describe('posts CRUD', () => {
  it('認証済みユーザーが投稿を作成できる', async () => {
    const ctx = testEnv.authenticatedContext('user-alice');
    await assertSucceeds(
      addDoc(collection(ctx.firestore(), 'posts'), {
        title:    'テスト',
        authorId: 'user-alice',
      })
    );
  });

  it('未認証ユーザーは投稿を作成できない', async () => {
    const ctx = testEnv.unauthenticatedContext();
    await assertFails(
      addDoc(collection(ctx.firestore(), 'posts'), { title: 'テスト' })
    );
  });

  it('作成者は自分の投稿を削除できる', async () => {
    // セキュリティルールを無効にして事前データを作成
    await testEnv.withSecurityRulesDisabled(async (ctx) => {
      await setDoc(doc(ctx.firestore(), 'posts', 'post-1'), {
        title: 'テスト', authorId: 'user-alice',
      });
    });

    const ctx = testEnv.authenticatedContext('user-alice');
    await assertSucceeds(deleteDoc(doc(ctx.firestore(), 'posts', 'post-1')));
  });

  it('他のユーザーの投稿は削除できない', async () => {
    await testEnv.withSecurityRulesDisabled(async (ctx) => {
      await setDoc(doc(ctx.firestore(), 'posts', 'post-1'), {
        title: 'テスト', authorId: 'user-alice',
      });
    });

    const ctx = testEnv.authenticatedContext('user-bob');
    await assertFails(deleteDoc(doc(ctx.firestore(), 'posts', 'post-1')));
  });
});
```

## 5. セキュリティルールのテスト

セキュリティルールのテストは [セキュリティルール詳細ガイド](09_セキュリティルール詳細ガイド.md) でも触れましたが、ここでは Storage ルールのテストを補足します。

```typescript
// tests/integration/storage.rules.test.ts
import {
  initializeTestEnvironment,
  assertSucceeds,
  assertFails,
  RulesTestEnvironment,
} from '@firebase/rules-unit-testing';
import { ref, uploadBytes, getDownloadURL, deleteObject } from 'firebase/storage';
import { readFileSync } from 'fs';

let testEnv: RulesTestEnvironment;

beforeAll(async () => {
  testEnv = await initializeTestEnvironment({
    projectId: 'demo-test',
    storage: {
      rules: readFileSync('storage.rules', 'utf8'),
      host: '127.0.0.1',
      port: 9199,
    },
  });
});

afterAll(async () => await testEnv.cleanup());

describe('Storage セキュリティルール', () => {
  const fakeFile = new Uint8Array([0x89, 0x50]); // ダミーバイト列

  it('本人はプロフィール画像をアップロードできる', async () => {
    const ctx = testEnv.authenticatedContext('user-alice');
    const fileRef = ref(ctx.storage(), 'users/user-alice/profile/avatar.jpg');
    await assertSucceeds(
      uploadBytes(fileRef, fakeFile, { contentType: 'image/jpeg' })
    );
  });

  it('他人のプロフィール画像はアップロードできない', async () => {
    const ctx = testEnv.authenticatedContext('user-bob');
    const fileRef = ref(ctx.storage(), 'users/user-alice/profile/avatar.jpg');
    await assertFails(
      uploadBytes(fileRef, fakeFile, { contentType: 'image/jpeg' })
    );
  });
});
```

```yaml
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with: { node-version: '20', cache: 'npm' }
      - uses: actions/setup-java@v4
        with: { distribution: 'temurin', java-version: '17' }
      - run: npm ci
      - run: npx playwright install --with-deps chromium
      - run: npm install -g firebase-tools
      - name: E2E テスト実行（エミュレーター起動 + Next.js 起動）
        run: |
          firebase emulators:start --only firestore,auth,functions &
          sleep 5
          npm run test:e2e
        env:
          NEXT_PUBLIC_USE_FIREBASE_EMULATOR: 'true'
          NEXT_PUBLIC_FIREBASE_PROJECT_ID: demo-test
```

## まとめ

| テスト種類 | 目的 | ツール | 実行タイミング |
|----------|------|-------|-------------|
| 単体テスト | ロジック・関数の検証 | Vitest + モック | 毎コミット |
| 統合テスト | Firebase との連携確認 | Vitest + Emulator | PR 作成時 |
| ルールテスト | セキュリティルールの検証 | rules-unit-testing | PR 作成時 |
| コンポーネントテスト | UI の挙動確認 | Testing Library | 毎コミット |
| E2E テスト | ユーザーシナリオの確認 | Playwright | PR 作成時・デプロイ前 |
