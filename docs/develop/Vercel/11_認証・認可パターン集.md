## 1. JWT（JSON Web Token）

### 1.1 JWT 基本実装

```typescript
// lib/jwt.ts
import jwt from 'jsonwebtoken';

const SECRET = process.env.JWT_SECRET || 'dev-secret';

// トークン生成
export function generateToken(payload: any, expiresIn = '7d') {
  return jwt.sign(payload, SECRET, { expiresIn });
}

// トークン検証
export function verifyToken(token: string) {
  try {
    return jwt.verify(token, SECRET) as any;
  } catch (error) {
    throw new Error('Invalid token');
  }
}

// トークンデコード（検証なし）
export function decodeToken(token: string) {
  return jwt.decode(token);
}
```

### 1.2 ペイロード設計

```typescript
// ✅ 推奨：最小限の情報

const token = generateToken(payload);

// ❌ 避ける：過剰な情報
const bloatedPayload = {

```

### 1.3 Refresh Token パターン

```typescript
// lib/tokens.ts
export interface TokenPair {
  accessToken: string;
  refreshToken: string;
}

export function generateTokenPair(userId: string): TokenPair {
  const accessToken = generateToken(
    { sub: userId, type: 'access' },
    '15m'  // 短命
  );

  const refreshToken = generateToken(
    { sub: userId, type: 'refresh' },
    '7d'  // 長命
  );

  return { accessToken, refreshToken };
}

// ページ/api/auth/refresh.ts
export default async function handler(req, res) {
  const { refreshToken } = req.body;

  try {
    const payload = verifyToken(refreshToken);

    if (payload.type !== 'refresh') {
      return res.status(401).json({ error: 'Invalid token type' });
    }

    const { accessToken, refreshToken: newRefreshToken } = 
      generateTokenPair(payload.sub);

    // 新しい refreshToken を DB に保存（オプション）
    await updateUserRefreshToken(payload.sub, newRefreshToken);

    res.json({ accessToken, refreshToken: newRefreshToken });
  } catch {
    res.status(401).json({ error: 'Invalid refresh token' });
  }
}
```

### 1.4 Cookie vs Header

```typescript
// ✅ Secure: HttpOnly Cookie (CSRF 対策必須)
export default function handler(req, res) {
  const token = generateToken({ sub: user.id });

  res.setHeader(
    'Set-Cookie',
    `token=${token}; HttpOnly; Secure; SameSite=Strict; Path=/; Max-Age=3600`
  );

  res.json({ success: true });
}

// ✅ 代替: Bearer Token (Authorization Header)
export default function handler(req, res) {
  const token = generateToken({ sub: user.id });
  res.setHeader('Authorization', `Bearer ${token}`);
  res.json({ token });
}

// ❌ 避ける: localStorage（XSS 脆弱性）
// クライアント: localStorage.setItem('token', token);
```

### 1.5 JWT Middleware

```typescript
// middleware/auth.ts
import { NextApiRequest, NextApiResponse } from 'next';
import { verifyToken } from '@/lib/jwt';

export function withJWT(handler: any) {
  return async (req: NextApiRequest, res: NextApiResponse) => {
    const token = req.headers.authorization?.split(' ')[1];

    if (!token) {
      return res.status(401).json({ error: 'No token' });
    }

    try {
      const payload = verifyToken(token);
      (req as any).user = payload;
      return handler(req, res);
    } catch {
      return res.status(401).json({ error: 'Invalid token' });
    }
  };
}

// pages/api/protected.ts
import { withJWT } from '@/middleware/auth';

export default withJWT(async (req, res) => {
  const user = (req as any).user;
  res.json({ message: `Hello ${user.email}` });
});
```

```typescript
// pages/api/auth/github/callback.ts
export default async function handler(req, res) {
  const { code, state } = req.query;

  // State 検証
  if (!verifyState(state as string)) {
    return res.status(400).json({ error: 'Invalid state' });
  }

  try {
    // トークン取得
    const tokenResponse = await fetch('https://github.com/login/oauth/access_token', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Accept': 'application/vnd.github.v3+json',
      },
      body: JSON.stringify({
        client_id: process.env.GITHUB_CLIENT_ID,
        client_secret: process.env.GITHUB_CLIENT_SECRET,
        code,
      }),
    });

    const { access_token } = await tokenResponse.json();

    // ユーザー情報取得
    const userResponse = await fetch('https://api.github.com/user', {
      headers: { Authorization: `token ${access_token}` },
    });

    const githubUser = await userResponse.json();

    // DB にユーザー保存
    const user = await upsertUser({
      githubId: githubUser.id,
      email: githubUser.email,
      name: githubUser.name,
      avatar: githubUser.avatar_url,
    });

    const token = generateToken({ sub: user.id });
    res.setHeader('Set-Cookie', 
      `token=${token}; HttpOnly; Secure; SameSite=Strict; Path=/`
    );

    res.redirect('/dashboard');
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
}
```

### 2.3 State Parameter（CSRF 対策）

```typescript
// lib/state.ts
import crypto from 'crypto';
import { kv } from '@vercel/kv';

export function generateState(): string {
  const state = crypto.randomBytes(32).toString('hex');
  // Redis に 10 分保存
  kv.setex(`oauth:state:${state}`, 600, '1');
  return state;
}

export async function verifyState(state: string): Promise<boolean> {
  const exists = await kv.get(`oauth:state:${state}`);
  if (exists) {
    await kv.del(`oauth:state:${state}`);
    return true;
  }
  return false;
}
```

## 3. next-auth.js 詳細

### 3.1 複数プロバイダ設定

```typescript
// pages/api/auth/[...nextauth].ts
import NextAuth from 'next-auth';
import GoogleProvider from 'next-auth/providers/google';
import GitHubProvider from 'next-auth/providers/github';
import CredentialsProvider from 'next-auth/providers/credentials';

export default NextAuth({
  providers: [
    GoogleProvider({
      clientId: process.env.GOOGLE_CLIENT_ID!,
      clientSecret: process.env.GOOGLE_CLIENT_SECRET!,
    }),
    GitHubProvider({
      clientId: process.env.GITHUB_CLIENT_ID!,
      clientSecret: process.env.GITHUB_CLIENT_SECRET!,
    }),
    CredentialsProvider({
      name: 'Credentials',
      credentials: {
        email: { label: 'Email', type: 'email' },
        password: { label: 'Password', type: 'password' },
      },
      async authorize(credentials) {
        const user = await db.user.findUnique({
          where: { email: credentials?.email },
        });

        if (!user) return null;

        const passwordMatch = await verifyPassword(
          credentials?.password!,
          user.password
        );

        return passwordMatch ? user : null;
      },
    }),
  ],

  callbacks: {
    async jwt({ token, user, account }) {
      if (user) {
        token.id = user.id;
        token.role = user.role;
      }
      return token;
    },

    async session({ session, token }) {
      session.user = {
        ...session.user,
        id: token.id as string,
        role: token.role as string,
      };
      return session;
    },
  },

  pages: {
    signIn: '/auth/signin',
    error: '/auth/error',
  },

  session: {
    strategy: 'jwt',
    maxAge: 7 * 24 * 60 * 60,  // 7日
  },

  secret: process.env.NEXTAUTH_SECRET,
});
```

## 4. セッション管理

### 4.1 Redis ベースセッション

```typescript
// lib/session.ts
import { kv } from '@vercel/kv';

export interface Session {
  userId: string;
  email: string;
  role: string;
  createdAt: number;
}

export async function createSession(user: any): Promise<string> {
  const sessionId = generateSessionId();
  const session: Session = {
    userId: user.id,
    email: user.email,
    role: user.role,
    createdAt: Date.now(),
  };

  // 24時間保存
  await kv.setex(`session:${sessionId}`, 86400, JSON.stringify(session));

  return sessionId;
}

export async function getSession(sessionId: string): Promise<Session | null> {
  const data = await kv.get(`session:${sessionId}`);
  return data ? JSON.parse(data) : null;
}

export async function destroySession(sessionId: string): Promise<void> {
  await kv.del(`session:${sessionId}`);
}
```

### 4.2 セッション検証 Middleware

```typescript
// middleware/session.ts
export function withSession(handler: any) {
  return async (req: NextApiRequest, res: NextApiResponse) => {
    const sessionId = req.cookies.sessionId;

    if (!sessionId) {
      return res.status(401).json({ error: 'Unauthorized' });
    }

    const session = await getSession(sessionId);
    if (!session) {
      return res.status(401).json({ error: 'Session expired' });
    }

    (req as any).session = session;
    return handler(req, res);
  };
}
```

### 4.3 セッション更新・延長

```typescript

}
```

## 5. ロールベース制御（RBAC）

### 5.1 ロール・権限定義

```typescript
// lib/rbac.ts
export enum Role {
  ADMIN = 'admin',
  MODERATOR = 'moderator',
  USER = 'user',
}

export const permissions = {
  [Role.ADMIN]: ['read', 'create', 'update', 'delete', 'manage_users'],
  [Role.MODERATOR]: ['read', 'create', 'update', 'moderate'],
  [Role.USER]: ['read', 'create'],
};

export function hasPermission(role: Role, permission: string): boolean {
  return permissions[role]?.includes(permission) ?? false;
}
```

### 5.2 RBAC Middleware

```typescript
// middleware/rbac.ts
export function withRole(...roles: Role[]) {
  return (handler: any) => {
    return async (req: NextApiRequest, res: NextApiResponse) => {
      const user = (req as any).user;

      if (!user || !roles.includes(user.role)) {
        return res.status(403).json({ error: 'Forbidden' });
      }

      return handler(req, res);
    };
  };
}

// 使用例
// pages/api/admin/users.ts
export default withRole(Role.ADMIN)(async (req, res) => {
  // ADMIN のみアクセス可能
  res.json({ users: [] });
});
```

### 5.3 属性ベース制御（ABAC）

```typescript
// lib/abac.ts
export interface Policy {
  effect: 'allow' | 'deny';
  resource: string;
  action: string;
  conditions?: Record<string, any>;
}

export const policies: Policy[] = [
  {
    effect: 'allow',
    resource: 'post',
    action: 'read',
  },
  {
    effect: 'allow',
    resource: 'post',
    action: 'create',
    conditions: { role: 'user' },
  },
  {
    effect: 'allow',
    resource: 'post',
    action: 'delete',
    conditions: { role: 'admin', isOwner: true },
  },
];

export function checkAccess(
  user: any,
  resource: string,
  action: string,
  context?: any
): boolean {
  return policies.some(policy => {
    if (policy.resource !== resource || policy.action !== action) {
      return false;
    }

    if (policy.conditions) {
      for (const [key, value] of Object.entries(policy.conditions)) {
        if (context?.[key] !== value) {
          return false;
        }
      }
    }

    return policy.effect === 'allow';
  });
}
```

## 6. Magic Link 認証

### 6.1 Magic Link 生成・送信

```typescript
// lib/magic-link.ts
import { kv } from '@vercel/kv';
import crypto from 'crypto';
import nodemailer from 'nodemailer';

export async function generateMagicLink(email: string): Promise<string> {
  const token = crypto.randomBytes(32).toString('hex');
  const expiresAt = Date.now() + 15 * 60 * 1000;  // 15分

  await kv.setex(
    `magic:${token}`,
    900,
    JSON.stringify({ email, expiresAt })
  );

  const link = `${process.env.VERCEL_URL}/auth/magic?token=${token}`;
  await sendEmail(email, link);

  return token;
}

async function sendEmail(email: string, link: string) {
  const transporter = nodemailer.createTransport({
    host: process.env.SMTP_HOST,
    port: process.env.SMTP_PORT,
    auth: {
      user: process.env.SMTP_USER,
      pass: process.env.SMTP_PASS,
    },
  });

  await transporter.sendMail({
    to: email,
    subject: 'Your Magic Link',
    html: `<a href="${link}">Sign in</a>`,
  });
}
```

### 6.2 Magic Link 検証・ログイン

```typescript
// pages/api/auth/magic.ts
export default async function handler(req, res) {
  const { token } = req.query;

  if (!token) {
    return res.status(400).json({ error: 'No token' });
  }

  const data = await kv.get(`magic:${token}`);

  if (!data) {
    return res.status(401).json({ error: 'Invalid or expired token' });
  }

  const { email } = JSON.parse(data);

  // ユーザー取得・作成
  let user = await db.user.findUnique({ where: { email } });
  if (!user) {
    user = await db.user.create({
      data: { email, name: email.split('@')[0] },
    });
  }

  // トークン削除
  await kv.del(`magic:${token}`);

  // セッション作成
  const sessionId = await createSession(user);

  res.setHeader('Set-Cookie',
    `sessionId=${sessionId}; HttpOnly; Secure; SameSite=Strict; Path=/; Max-Age=86400`
  );

  res.redirect('/dashboard');
}
```

## 7. 2FA / TOTP

### 7.1 TOTP 登録

```bash
npm install speakeasy qrcode
```

```typescript
// lib/totp.ts
import speakeasy from 'speakeasy';
import QRCode from 'qrcode';

export async function generateTOTPSecret(email: string) {
  const secret = speakeasy.generateSecret({
    name: `MyApp (${email})`,
    issuer: 'MyApp',
    length: 32,
  });

  const qrCode = await QRCode.toDataURL(secret.otpauth_url!);

  return {
    secret: secret.base32,
    qrCode,
  };
}

export function verifyTOTP(secret: string, token: string): boolean {
  return speakeasy.totp.verify({
    secret,
    encoding: 'base32',
    token,
    window: 2,  // ±2 ウィンドウ
  });
}
```

```typescript
  const { secret } = await generateTOTPSecret(user.email);

// pages/api/auth/totp/verify.ts
export default async function handler(req, res) {
  const user = (req as any).user;
  const { token } = req.body;

  const secret = await kv.get(`totp:verify:${user.id}`);
  if (!secret) {
    return res.status(400).json({ error: 'Setup expired' });
  }

  if (!verifyTOTP(secret, token)) {
    return res.status(401).json({ error: 'Invalid token' });
  }

  // DB に保存
  await db.user.update({
    where: { id: user.id },
    data: { totpSecret: secret, totpEnabled: true },
  });

  await kv.del(`totp:verify:${user.id}`);

  res.json({ success: true });
}
```

### 7.3 TOTP ログイン検証

```typescript
// pages/api/auth/login-totp.ts
export default async function handler(req, res) {
  const { email, password, totpToken } = req.body;

  // ユーザー認証
  const user = await db.user.findUnique({ where: { email } });
  if (!user || !verifyPassword(password, user.password)) {
    return res.status(401).json({ error: 'Invalid credentials' });
  }

  // TOTP 有効な場合検証
  if (user.totpEnabled) {
    if (!totpToken || !verifyTOTP(user.totpSecret, totpToken)) {
      return res.status(401).json({ error: 'Invalid TOTP token' });
    }
  }

  const token = generateToken({ sub: user.id });
  res.json({ token });
}
```

## 8. トークンリフレッシュ戦略

### 8.1 Sliding Window Strategy

```typescript
// lib/token-strategy.ts
const ACCESS_TOKEN_LIFETIME = 15 * 60;  // 15分
const REFRESH_TOKEN_LIFETIME = 7 * 24 * 60 * 60;  // 7日
const REFRESH_THRESHOLD = 5 * 60;  // 5分

export function shouldRefreshToken(iat: number): boolean {
  const age = Math.floor(Date.now() / 1000) - iat;
  return age > ACCESS_TOKEN_LIFETIME - REFRESH_THRESHOLD;
}
```

```typescript
// pages/api/protected.ts
export default withJWT(async (req, res) => {
  const user = (req as any).user;

  // トークンが期限切れ間近？
  if (shouldRefreshToken(user.iat)) {
    const newToken = generateToken({ sub: user.id });
    res.setHeader('X-New-Token', newToken);
  }

  res.json({ data: 'protected' });
});
```

### 8.2 Rotation Strategy

```typescript
// Refresh Token をリクエストごとに回転
export default async function handler(req, res) {
  const { refreshToken } = req.body;

  try {
    const payload = verifyToken(refreshToken);
    
    // 古い refreshToken を無効化
    await kv.del(`refreshtoken:${payload.sub}:${payload.jti}`);

    // 新規トークンペア生成
    const jti = generateRandomId();
    const newAccessToken = generateToken({
      sub: payload.sub,
      type: 'access',
    }, '15m');

    const newRefreshToken = generateToken({
      sub: payload.sub,
      type: 'refresh',
      jti,
    }, '7d');

    // 新規 refresh token を記録
    await kv.setex(`refreshtoken:${payload.sub}:${jti}`, 604800, '1');

    res.json({
      accessToken: newAccessToken,
      refreshToken: newRefreshToken,
    });
  } catch {
    res.status(401).json({ error: 'Invalid token' });
  }
}
```
