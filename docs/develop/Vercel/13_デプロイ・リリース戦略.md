## 2. Blue-Green デプロイ

### 2.1 Vercel での Blue-Green 実装

**基本戦略:**

```
1. 現在 (Blue): my-app.vercel.app → Deployment A
2. 新規 (Green): staging.my-app.vercel.app → Deployment B (新バージョン)
3. テスト: Green で完全検証
4. 切り替え: my-app.vercel.app → Deployment B
5. ロールバック: my-app.vercel.app → Deployment A (もし必要)
```

### 2.2 実装手順

**ステップ 1: 環境構築**

```bash
# .github/workflows/blue-green.yml
name: Blue-Green Deployment

on:
  push:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v3
        with:
          node-version: 18
      - run: npm ci
      - run: npm test
      - run: npm run build

  deploy-green:
    needs: test
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - uses: vercel/action@v1
        with:
          vercel-token: ${{ secrets.VERCEL_TOKEN }}
          vercel-org-id: ${{ secrets.VERCEL_ORG_ID }}
          vercel-project-id: ${{ secrets.VERCEL_PROJECT_ID }}
          # プロダクション環境にはデプロイしない
          production: false
    outputs:
      deployment-url: ${{ steps.deploy.outputs.preview-url }}

  smoke-test:
    needs: deploy-green
    runs-on: ubuntu-latest
    steps:
      - name: Run smoke tests
        run: |
          curl -f ${{ needs.deploy-green.outputs.deployment-url }}/api/health
          echo "✅ Green environment healthy"

  switch-traffic:
    needs: smoke-test
    runs-on: ubuntu-latest
    if: success()
    steps:
      - name: Switch to Green
        run: |
          npm install -g vercel
          vercel promote ${{ needs.deploy-green.outputs.deployment-url }} \
            --token ${{ secrets.VERCEL_TOKEN }}
          echo "✅ Switched to Green deployment"

  cleanup:
    needs: switch-traffic
    runs-on: ubuntu-latest
    if: success()
    steps:
      - name: Keep last 3 deployments
        run: |
          echo "Old Blue deployment can be deleted manually"
```

### 2.3 ロールバック スクリプト

```bash
#!/bin/bash
# scripts/rollback-blue-green.sh

# 前回のデプロイに戻す
PREVIOUS_DEPLOYMENT=$(vercel list --limit=2 --token=$VERCEL_TOKEN | tail -1 | cut -d' ' -f1)

vercel promote $PREVIOUS_DEPLOYMENT --token=$VERCEL_TOKEN

echo "✅ Rolled back to $PREVIOUS_DEPLOYMENT"
```

## 3. Canary デプロイ

### 3.1 トラフィック段階的切り替え

```typescript
// middleware.ts - Canary Traffic Split
import { NextResponse } from 'next/server';

interface CanaryConfig {
  enabled: boolean;
  percentage: number;  // 0-100
  deploymentId: string;
}

const canaryConfig: CanaryConfig = {
  enabled: true,
  percentage: 10,  // 10% のユーザーに新バージョン
  deploymentId: 'vercel-deployment-xyz',
};

function isCanaryUser(userId: string): boolean {
  // ユーザーID のハッシュで決定的に割り当て
  const hash = parseInt(userId.substring(0, 8), 16);
  return (hash % 100) < canaryConfig.percentage;
}

export function middleware(request: NextRequest) {
  if (!canaryConfig.enabled) {
    return NextResponse.next();
  }

  const userId = request.cookies.get('user-id')?.value;

  if (userId && isCanaryUser(userId)) {
    // Canary 環境へ
    return NextResponse.rewrite(
      new URL(`https://canary-${canaryConfig.deploymentId}.vercel.app${request.nextUrl.pathname}`, request.url)
    );
  }

  return NextResponse.next();
}
```

### 3.2 Canary 監視・メトリクス

```typescript
// pages/api/canary-metrics.ts
import { kv } from '@vercel/kv';

export default async function handler(req, res) {
  const { environment } = req.query;  // 'canary' or 'stable'

  if (req.method === 'POST') {
    const { metric, value } = req.body;

    await kv.hincrbyfloat(
      `canary:metrics:${environment}`,
      metric,
      value
    );

    return res.json({ success: true });
  }

  // メトリクス取得
  const metrics = await kv.hgetall(`canary:metrics:${environment}`);
  res.json(metrics);
}
```

### 3.3 Canary デプロイ自動化

```bash
# .github/workflows/canary.yml
name: Canary Deployment

on:
  push:
    branches: [canary]

jobs:
  deploy-canary:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - uses: vercel/action@v1
        with:
          vercel-token: ${{ secrets.VERCEL_TOKEN }}
          production: false
        id: deploy

      - name: Monitor Canary (30 min)
        run: |
          echo "🔍 Monitoring canary deployment for 30 minutes"
          sleep 1800
          
          # メトリクス確認
          CANARY_ERRORS=$(curl ${{ steps.deploy.outputs.preview-url }}/api/canary-metrics?environment=canary)
          echo "Canary errors: $CANARY_ERRORS"

      - name: Evaluate Canary Success
        run: |
          # エラー率が 0.1% 以下なら本番環境へプロモート
          if [ $(echo "$CANARY_ERRORS < 0.1" | bc) -eq 1 ]; then
            vercel promote ${{ steps.deploy.outputs.preview-url }} \
              --token ${{ secrets.VERCEL_TOKEN }}
            echo "✅ Canary promoted to production"
          else
            echo "❌ Canary failed - keeping on staging"
          fi
```

## 5. A/B テスト環境構築

### 5.1 A/B テスト フレームワーク

```typescript
// lib/ab-test.ts
import { kv } from '@vercel/kv';

export interface ABTest {
  id: string;
  variants: {
    control: string;
    treatment: string;
  };
  metrics: {
    conversions: number;
    views: number;
  };
  status: 'active' | 'paused' | 'completed';
}

export async function assignVariant(
  testId: string,
  userId: string
): Promise<'control' | 'treatment'> {
  // ユーザー ID のハッシュで決定的に割り当て
  const hash = parseInt(userId.substring(0, 8), 16);
  return hash % 2 === 0 ? 'control' : 'treatment';
}

export async function trackConversion(testId: string, variant: string) {
  await kv.hincrby(`ab:${testId}`, `conversions:${variant}`, 1);
}

export async function getResults(testId: string) {
  const data = await kv.hgetall(`ab:${testId}`);
  
  const controlConversions = parseInt(data['conversions:control'] || '0');
  const treatmentConversions = parseInt(data['conversions:treatment'] || '0');

  return {
    controlConversionRate: controlConversions / parseInt(data['views:control'] || '1'),
    treatmentConversionRate: treatmentConversions / parseInt(data['views:treatment'] || '1'),
    winner: controlConversions > treatmentConversions ? 'control' : 'treatment',
  };
}
```

### 5.2 A/B テストコンポーネント

```jsx
// components/ABTestWrapper.tsx
import { useEffect, useState } from 'react';
import { assignVariant } from '@/lib/ab-test';

interface Props {
  testId: string;
  userId: string;
  control: React.ReactNode;
  treatment: React.ReactNode;
}

export default function ABTestWrapper({
  testId,
  userId,
  control,
  treatment,
}: Props) {
  const [variant, setVariant] = useState<'control' | 'treatment' | null>(null);

  useEffect(() => {
    async function init() {
      const v = await assignVariant(testId, userId);
      setVariant(v);

      // View を記録
      await fetch(`/api/ab-test/track`, {
        method: 'POST',
        body: JSON.stringify({ testId, variant: v, event: 'view' }),
      });
    }

    init();
  }, [testId, userId]);

  if (!variant) return <div>Loading...</div>;

  return variant === 'control' ? control : treatment;
}
```

### 5.3 統計分析

```typescript
// lib/ab-statistics.ts
export function calculateConfidenceInterval(
  conversions: number,
  views: number,
  zScore = 1.96  // 95% confidence
) {
  const conversionRate = conversions / views;
  const standardError = Math.sqrt(
    (conversionRate * (1 - conversionRate)) / views
  );

  return {
    conversionRate,
    lower: conversionRate - zScore * standardError,
    upper: conversionRate + zScore * standardError,
    marginOfError: zScore * standardError,
  };
}

export function isSignificant(
  controlConversions: number,
  controlViews: number,
  treatmentConversions: number,
  treatmentViews: number
): boolean {
  const controlCI = calculateConfidenceInterval(controlConversions, controlViews);
  const treatmentCI = calculateConfidenceInterval(treatmentConversions, treatmentViews);

  // 信頼区間が重ならない = 統計的に有意
  return controlCI.upper < treatmentCI.lower || treatmentCI.upper < controlCI.lower;
}
```

```bash
# .github/workflows/release.yml
name: Release

on:
  push:
    branches: [main]
    paths:
      - 'src/**'
      - 'package.json'

jobs:
  release:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
        with:
          fetch-depth: 0

      - name: Generate Changelog
        uses: conventional-changelog/standard-version@v9
        with:
          release-as: minor

      - name: Create Release
        uses: actions/create-release@v1
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        with:
          tag_name: ${{ env.new_version }}
          release_name: Release ${{ env.new_version }}
          body_path: CHANGELOG.md

      - name: Deploy
        uses: vercel/action@v1
        with:
          vercel-token: ${{ secrets.VERCEL_TOKEN }}
          production: true
```

## 7. ロールバック戦術

### 7.1 即座ロールバック（1 分以内）

```bash
# 方法 1: Vercel CLI で前回デプロイメントに戻す
vercel promote [PREVIOUS_DEPLOYMENT_URL] --token $VERCEL_TOKEN

# 方法 2: Git で前のコミットにリセット
git revert HEAD
git push origin main
# Vercel が自動デプロイ（3-5 分）

# 方法 3: Vercel ダッシュボード
# Project > Deployments > [前のデプロイメント] > Promote to Production
```

### 7.2 段階的ロールバック

```typescript
// middleware.ts - Gradual Rollback
export function middleware(request: NextRequest) {
  const rollbackPercentage = 20;  // 20% は前バージョン

  const userId = request.cookies.get('user-id')?.value || '';
  const hash = parseInt(userId.substring(0, 8), 16);

  if ((hash % 100) < rollbackPercentage) {
    // 前バージョンのサーバーへ
    return NextResponse.rewrite(
      new URL(`https://stable.my-app.vercel.app${request.nextUrl.pathname}`, request.url)
    );
  }

  return NextResponse.next();
}
```
