```typescript
import * as go from "gojs";

function createOrgChart(divId: string): go.Diagram {
  const diagram = new go.Diagram(divId, {
    "undoManager.isEnabled": true,
    layout: new go.TreeLayout({
      angle: 90,           // 上から下
      layerSpacing: 40,
      nodeSpacing: 20,
      treeStyle: go.TreeStyle.LastParents,
      alternateAngle: 90,
      alternateLayerSpacing: 40,
    }),
  });

  // ━━━ ノードテンプレート ━━━
  diagram.nodeTemplate = new go.Node("Auto", {
    doubleClick: (e, node) => {
      // ダブルクリックでサブツリーを折りたたみ
      const n = node as go.Node;
      if (n.isTreeExpanded) {
        diagram.commandHandler.collapseTree(n);
      } else {
        diagram.commandHandler.expandTree(n);
      }
    },
  }).add(
    new go.Shape("RoundedRectangle", {
      fill: "#2196F3",
      stroke: "#1565C0",
      strokeWidth: 1,
    }).bind("fill", "color"),

    new go.Panel("Table", { defaultAlignment: go.Spot.Left, margin: 4 }).add(
      // 名前（1行目）
      new go.TextBlock({
        row: 0,
        column: 0,
        columnSpan: 2,
        margin: new go.Margin(4, 8),
        font: "bold 14px sans-serif",
        stroke: "white",
        maxSize: new go.Size(160, NaN),
        wrap: go.Wrap.Fit,
      }).bind("text", "name"),

      // 役職（2行目）
      new go.TextBlock({
        row: 1,
        column: 0,
        columnSpan: 2,
        margin: new go.Margin(2, 8),
        font: "12px sans-serif",
        stroke: "rgba(255,255,255,0.8)",
      }).bind("text", "title")

  // ━━━ リンクテンプレート ━━━

    routing: go.Routing.Orthogonal,
    corner: 5,
    selectable: false,

  );
```

## 2. フローチャート（Flowchart）

```typescript
import * as go from "gojs";

function createFlowchart(divId: string): go.Diagram {
  const diagram = new go.Diagram(divId, {
    "undoManager.isEnabled": true,
    // 層状有向グラフレイアウト（上→下にフロー）
    layout: new go.LayeredDigraphLayout({
      direction: 90,
      layerSpacing: 40,
      columnSpacing: 30,
    }),
  });

  // ━━━ ノードテンプレートマップ ━━━
  // 複数テンプレートを category で切り替え
  diagram.nodeTemplateMap = new go.Map<string, go.Node>();

  // デフォルト（処理）
  diagram.nodeTemplateMap.add("",
    new go.Node("Auto").add(
      new go.Shape("Rectangle", {
        fill: "#4CAF50", stroke: "#2E7D32", strokeWidth: 1.5,
      }),
      new go.TextBlock({
        margin: 10, font: "14px sans-serif", stroke: "white",
      }).bind("text")
    )
  );

  // 開始・終了（角丸）
  diagram.nodeTemplateMap.add("Terminal",
    new go.Node("Auto").add(
      new go.Shape("RoundedRectangle", {
        fill: "#FF5722", stroke: "#BF360C", strokeWidth: 1.5,
        parameter1: 20, // 角丸の半径
      }),
      new go.TextBlock({
        margin: 10, font: "bold 14px sans-serif", stroke: "white",
      }).bind("text")
    )
  );

  // 条件分岐（ひし形）
  diagram.nodeTemplateMap.add("Decision",
    new go.Node("Auto").add(
      new go.Shape("Diamond", {
        fill: "#FFC107", stroke: "#F57F17", strokeWidth: 1.5,
        desiredSize: new go.Size(120, 80),
      }),
      new go.TextBlock({
        margin: 8, font: "13px sans-serif", stroke: "#333",
        textAlign: "center",
      }).bind("text")
    )
  );

  // 入出力（平行四辺形）
  diagram.nodeTemplateMap.add("IO",
    new go.Node("Auto").add(
      new go.Shape("Parallelogram1", {
        fill: "#2196F3", stroke: "#1565C0", strokeWidth: 1.5,
      }),
      new go.TextBlock({
        margin: 10, font: "14px sans-serif", stroke: "white",
      }).bind("text")
    )
  );

  // ━━━ リンクテンプレート ━━━
  diagram.linkTemplate = new go.Link({
    routing: go.Routing.AvoidsNodes,
    corner: 5,
    curve: go.Curve.JumpOver,
  }).add(
    new go.Shape({ strokeWidth: 2, stroke: "#666" }),
    new go.Shape({ toArrow: "Standard", fill: "#666", stroke: null }),
    // リンク上のラベル（Yes / No 等）
    new go.Panel("Auto").add(
      new go.Shape("RoundedRectangle", {
        fill: "white", stroke: "#ccc",
      }),
      new go.TextBlock({
        margin: 3, font: "12px sans-serif",
      }).bind("text", "label")
    )
  );

  // ━━━ モデルデータ ━━━
  diagram.model = new go.GraphLinksModel(
    // ノードデータ
    [
      { key: 1, text: "開始",           category: "Terminal" },
      { key: 2, text: "ユーザー入力",   category: "IO" },
      { key: 3, text: "バリデーション", },
      { key: 4, text: "有効？",         category: "Decision" },
      { key: 5, text: "データ保存",     },
      { key: 6, text: "エラー表示",     },
      { key: 7, text: "完了画面表示",   category: "IO" },
      { key: 8, text: "終了",           category: "Terminal" },
    ],
    // リンクデータ
    [
      { from: 1, to: 2 },
      { from: 2, to: 3 },
      { from: 3, to: 4 },
      { from: 4, to: 5, label: "Yes" },
      { from: 4, to: 6, label: "No" },
      { from: 5, to: 7 },
      { from: 6, to: 2 },
      { from: 7, to: 8 },
    ]
  );

  return diagram;
}
```

```typescript
import * as go from "gojs";

function createMindMap(divId: string): go.Diagram {

    "undoManager.isEnabled": true,

      angle: 0,            // 左→右
      layerSpacing: 80,
      nodeSpacing: 10,
      arrangement: go.TreeArrangement.FixedRoots,
    }),
    // 背景グリッド
    "grid.visible": true,
  });

  // ━━━ ノードテンプレート ━━━
  diagram.nodeTemplate = new go.Node("Horizontal", {
    selectionAdornmentTemplate:
      new go.Adornment("Auto").add(
        new go.Shape("RoundedRectangle", {
          fill: null, stroke: "#2196F3", strokeWidth: 2,

        new go.Placeholder()

    // カラーバー
```

```typescript
import * as go from "gojs";

function createNetworkDiagram(divId: string): go.Diagram {
  const diagram = new go.Diagram(divId, {
    "undoManager.isEnabled": true,
    // 力学モデルレイアウト（ノードが自然に広がる）
    layout: new go.ForceDirectedLayout({
      maxIterations: 300,
      defaultSpringLength: 80,
      defaultElectricalCharge: 100,
    }),
  });

  // ━━━ ノードテンプレート ━━━
  diagram.nodeTemplate = new go.Node("Vertical").add(
    // アイコン（図形で代用）
    new go.Shape("Circle", {
      desiredSize: new go.Size(50, 50),
      strokeWidth: 2,
    })
      .bind("fill", "color")
      .bind("figure", "shape"),

    // ラベル
    new go.TextBlock({
      margin: new go.Margin(5, 0, 0, 0),
      font: "12px sans-serif",
    }).bind("text", "name"),

    // IP アドレス
    new go.TextBlock({
      font: "10px monospace",
      stroke: "#666",
    }).bind("text", "ip")
  );

  // ━━━ リンクテンプレート ━━━
  diagram.linkTemplate = new go.Link().add(
    new go.Shape({ strokeWidth: 2, stroke: "#999" }),
    // 帯域幅ラベル
    new go.TextBlock({
      segmentOffset: new go.Point(0, -10),
      font: "10px sans-serif",
      stroke: "#666",
    }).bind("text", "bandwidth")
  );

  // ━━━ モデルデータ ━━━
  diagram.model = new go.GraphLinksModel(
    [
```

## 5. ER 図（Entity Relationship）

```typescript
import * as go from "gojs";

function createERDiagram(divId: string): go.Diagram {
  const diagram = new go.Diagram(divId, {
    "undoManager.isEnabled": true,
    layout: new go.ForceDirectedLayout({
      maxIterations: 200,
      defaultSpringLength: 100,
    }),
  });

  // ━━━ ノードテンプレート（テーブル） ━━━
  diagram.nodeTemplate = new go.Node("Auto", {
    locationSpot: go.Spot.Center,
  }).add(
    new go.Shape("RoundedRectangle", {
      fill: "white",
      stroke: "#333",
      strokeWidth: 1.5,
    }),
    new go.Panel("Table", {
      defaultRowSeparatorStroke: "#ddd",
      defaultColumnSeparatorStroke: "#ddd",
    }).add(
      // テーブル名（ヘッダ）
      new go.TextBlock({
        row: 0, column: 0, columnSpan: 3,
        margin: new go.Margin(6, 10),
        font: "bold 14px sans-serif",
        alignment: go.Spot.Center,
        background: "#2196F3",
        stroke: "white",
        width: 200,
      }).bind("text", "name"),

      // カラム一覧
      new go.Panel("Vertical", {
        row: 1, column: 0, columnSpan: 3,
        alignment: go.Spot.TopLeft,
        defaultAlignment: go.Spot.Left,
        margin: 4,
        itemTemplate: new go.Panel("Horizontal", { margin: 2 }).add(
          // PK アイコン
          new go.TextBlock({
            font: "10px sans-serif",
            stroke: "#FF9800",
            width: 20,
          }).bind("text", "pk", (pk: boolean) => pk ? "🔑" : ""),
          // カラム名
          new go.TextBlock({
            font: "13px sans-serif",
            width: 100,
          }).bind("text", "name")
            .bind("font", "pk", (pk: boolean) =>
              pk ? "bold 13px sans-serif" : "13px sans-serif"
            ),
          // 型
          new go.TextBlock({
            font: "12px monospace",
            stroke: "#666",
          }).bind("text", "type"),
        ),
      }).bind("itemArray", "columns")
    )
  );

  // ━━━ リンクテンプレート ━━━
  diagram.linkTemplate = new go.Link({
    routing: go.Routing.Orthogonal,
    corner: 5,
    relinkableFrom: true,
    relinkableTo: true,
  }).add(
    new go.Shape({ strokeWidth: 2, stroke: "#666" }),
    // from 側のラベル（1 or *）
    new go.TextBlock({
      segmentIndex: 0,
      segmentOffset: new go.Point(NaN, NaN),
      segmentOrientation: go.Orientation.Upright,
      font: "bold 12px sans-serif",
    }).bind("text", "fromCardinality"),
    // to 側のラベル
    new go.TextBlock({
      segmentIndex: -1,
      segmentOffset: new go.Point(NaN, NaN),
      segmentOrientation: go.Orientation.Upright,
      font: "bold 12px sans-serif",
    }).bind("text", "toCardinality"),
  );

  // ━━━ モデルデータ ━━━
  diagram.model = new go.GraphLinksModel(
    [
      {
        key: "users", name: "Users",
        columns: [
          { name: "id",         type: "INT",     pk: true },
          { name: "email",      type: "VARCHAR", pk: false },
          { name: "name",       type: "VARCHAR", pk: false },
          { name: "created_at", type: "DATETIME", pk: false },
        ],
      },
      {
        key: "posts", name: "Posts",
        columns: [
          { name: "id",        type: "INT",     pk: true },
          { name: "title",     type: "VARCHAR", pk: false },
          { name: "content",   type: "TEXT",    pk: false },
          { name: "author_id", type: "INT",     pk: false },
        ],
      },
      {
        key: "tags", name: "Tags",
        columns: [
          { name: "id",   type: "INT",     pk: true },
          { name: "name", type: "VARCHAR", pk: false },
        ],
      },
      {
        key: "post_tags", name: "Post_Tags",
        columns: [
          { name: "post_id", type: "INT", pk: true },
          { name: "tag_id",  type: "INT", pk: true },
        ],
      },
    ],
    [
      { from: "users",     to: "posts",     fromCardinality: "1", toCardinality: "*" },
      { from: "posts",     to: "post_tags", fromCardinality: "1", toCardinality: "*" },
      { from: "tags",      to: "post_tags", fromCardinality: "1", toCardinality: "*" },
    ]
  );

  return diagram;
}
```

```typescript
import * as go from "gojs";

function createStateChart(divId: string): go.Diagram {
  const diagram = new go.Diagram(divId, {
    "undoManager.isEnabled": true,
    layout: new go.LayeredDigraphLayout({
      direction: 0,
      layerSpacing: 80,
      columnSpacing: 40,
    }),
  });

  // ━━━ 状態ノード ━━━
  diagram.nodeTemplateMap.add("",
    new go.Node("Auto").add(
      new go.Shape("RoundedRectangle", {
        fill: "#E3F2FD",
        stroke: "#1976D2",
        strokeWidth: 2,
        parameter1: 10,
      }).bind("fill", "color"),
      new go.TextBlock({
        margin: 12,
        font: "bold 14px sans-serif",
        stroke: "#333",
      }).bind("text")
    )
  );

  // 開始状態（黒丸）
  diagram.nodeTemplateMap.add("Start",
    new go.Node("Spot").add(
      new go.Shape("Circle", {
        desiredSize: new go.Size(30, 30),
        fill: "#333",
        stroke: null,
      })
    )
  );

  // 終了状態（二重丸）
  diagram.nodeTemplateMap.add("End",
    new go.Node("Spot").add(
      new go.Shape("Circle", {
        desiredSize: new go.Size(36, 36),
        fill: "white",
        stroke: "#333",
        strokeWidth: 2,
      }),
      new go.Shape("Circle", {
        desiredSize: new go.Size(24, 24),
        fill: "#333",
        stroke: null,
      })
    )
  );

  // ━━━ 遷移（リンク） ━━━
  diagram.linkTemplate = new go.Link({
    curve: go.Curve.Bezier,
    adjusting: go.LinkAdjusting.Stretch,
    toShortLength: 3,
  }).add(
    new go.Shape({ strokeWidth: 2, stroke: "#555" }),
    new go.Shape({ toArrow: "Standard", fill: "#555", stroke: null }),
    // 遷移条件ラベル
    new go.Panel("Auto").add(
      new go.Shape("RoundedRectangle", {
        fill: "#FFFDE7",
        stroke: "#FBC02D",
      }),
      new go.TextBlock({
        margin: 4,
        font: "12px sans-serif",
      }).bind("text", "event")
    )
  );

  // ━━━ モデルデータ ━━━
  diagram.model = new go.GraphLinksModel(
    [
      { key: "start",     category: "Start" },
      { key: "idle",      text: "待機中",   color: "#E3F2FD" },
      { key: "loading",   text: "読込中",   color: "#FFF9C4" },
      { key: "success",   text: "成功",     color: "#C8E6C9" },
      { key: "error",     text: "エラー",   color: "#FFCDD2" },
      { key: "end",       category: "End" },
    ],
    [
      { from: "start",   to: "idle",    event: "初期化" },
      { from: "idle",    to: "loading", event: "リクエスト送信" },
      { from: "loading", to: "success", event: "レスポンス受信" },
      { from: "loading", to: "error",   event: "タイムアウト" },
      { from: "success", to: "idle",    event: "リセット" },
      { from: "error",   to: "loading", event: "リトライ" },
      { from: "error",   to: "idle",    event: "キャンセル" },
      { from: "success", to: "end",     event: "完了" },
```

```typescript
function createKanban(divId: string): go.Diagram {

  // ━━━ グループテンプレート（カラム: TODO / Doing / Done） ━━━

      wrappingColumn: 1,
      cellSize: new go.Size(220, NaN),
      spacing: new go.Size(0, 8),
    }),
    selectable: false,
```

## 8. CRUD 操作（ノードの追加・更新・削除）

```typescript
import * as go from "gojs";

// 前提: diagram は既に初期化済み
declare const diagram: go.Diagram;

// ━━━ CREATE: ノードを追加 ━━━
function addNode(name: string, parentKey?: number): void {
  diagram.startTransaction("add node");

  const newKey = diagram.model.nodeDataArray.length + 1;
  const nodeData: any = { key: newKey, name };

  if (diagram.model instanceof go.TreeModel && parentKey !== undefined) {
    nodeData.parent = parentKey;
  }

  diagram.model.addNodeData(nodeData);
  diagram.commitTransaction("add node");
}

// ━━━ READ: ノードデータを取得 ━━━
function getNodeData(key: number): go.ObjectData | null {
  return diagram.model.findNodeDataForKey(key);
}

// 全ノードデータを取得
function getAllNodes(): go.ObjectData[] {
  return diagram.model.nodeDataArray;
}

// ━━━ UPDATE: ノードのプロパティを更新 ━━━
function updateNode(key: number, props: Record<string, any>): void {
  diagram.startTransaction("update node");

  const nodeData = diagram.model.findNodeDataForKey(key);
  if (nodeData) {
    for (const [prop, value] of Object.entries(props)) {
      diagram.model.set(nodeData, prop, value);
    }
  }

  diagram.commitTransaction("update node");
}

// ━━━ DELETE: ノードを削除 ━━━
function deleteNode(key: number): void {
  diagram.startTransaction("delete node");

  const nodeData = diagram.model.findNodeDataForKey(key);
  if (nodeData) {
    diagram.model.removeNodeData(nodeData);
  }

  diagram.commitTransaction("delete node");
}

// ━━━ リンクの追加（GraphLinksModel の場合） ━━━
function addLink(fromKey: number, toKey: number, label?: string): void {
  diagram.startTransaction("add link");

  if (diagram.model instanceof go.GraphLinksModel) {
    const linkData: any = { from: fromKey, to: toKey };
    if (label) linkData.label = label;
    diagram.model.addLinkData(linkData);
  }

  diagram.commitTransaction("add link");
}

// ━━━ 使用例 ━━━
addNode("新しいノード", 1);
updateNode(2, { name: "更新されたノード", color: "#FF5722" });
deleteNode(5);
```

> **⚠️ 重要:** モデルデータの変更は必ず `startTransaction()` / `commitTransaction()` で囲んでください。Undo/Redo が正しく動作するために必要です。

パレットから要素をダイアグラムにドラッグ＆ドロップできます。

```typescript
import * as go from "gojs";

function setupPalette(
  paletteDivId: string,
  diagramDivId: string
): { palette: go.Palette; diagram: go.Diagram } {

  // ━━━ メインダイアグラム ━━━
  const diagram = new go.Diagram(diagramDivId, {
    "undoManager.isEnabled": true,
    allowDrop: true,  // ← ドロップを許可
  });

  diagram.nodeTemplate = new go.Node("Auto").add(
    new go.Shape("RoundedRectangle", {
      fill: "white", stroke: "#333", strokeWidth: 1.5,
    }).bind("fill", "color"),
    new go.TextBlock({ margin: 10, font: "14px sans-serif" })
      .bind("text")
  );

  // ━━━ パレット ━━━
  const palette = new go.Palette(paletteDivId, {
    // パレット用の同じノードテンプレートを使用
    nodeTemplate: diagram.nodeTemplate.copy(),
  });

  // パレットに表示する要素
  palette.model = new go.Model([
    { text: "処理",     color: "#4CAF50" },
    { text: "条件分岐", color: "#FFC107" },
    { text: "入出力",   color: "#2196F3" },
    { text: "開始/終了", color: "#FF5722" },
  ]);

  return { palette, diagram };
}
```

## 10. JSON 保存・復元

```typescript
import * as go from "gojs";

declare const diagram: go.Diagram;

// ━━━ 保存 ━━━
function saveDiagram(): string {
  return diagram.model.toJson();
}

// ━━━ 復元 ━━━
function loadDiagram(jsonString: string): void {
  diagram.model = go.Model.fromJson(jsonString);
}

// ━━━ ローカルストレージに保存 ━━━
function saveToLocalStorage(): void {
  const json = diagram.model.toJson();
  localStorage.setItem("myDiagram", json);
  console.log("✅ 保存しました");
}

// ━━━ ローカルストレージから復元 ━━━
function loadFromLocalStorage(): void {
  const json = localStorage.getItem("myDiagram");
  if (json) {
    diagram.model = go.Model.fromJson(json);
    console.log("✅ 復元しました");
  }
}

// ━━━ サーバーに保存（fetch） ━━━
async function saveToServer(): Promise<void> {
  const json = diagram.model.toJson();
  await fetch("/api/diagram", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ data: json }),
  });
}

// ━━━ サーバーから読み込み ━━━
async function loadFromServer(id: string): Promise<void> {
  const res = await fetch(`/api/diagram/${id}`);
  const { data } = await res.json();
  diagram.model = go.Model.fromJson(data);
}
```

## 11. 画像エクスポート（SVG / PNG）

```typescript
import * as go from "gojs";

declare const diagram: go.Diagram;

// ━━━ SVG として取得 ━━━
function exportSVG(): string {
  const svg = diagram.makeSvg({
    scale: 1,
    background: "white",
  });
  if (!svg) return "";
  return new XMLSerializer().serializeToString(svg);
}

// ━━━ PNG としてダウンロード ━━━
function downloadPNG(): void {
  const img = diagram.makeImage({
    scale: 2,         // 解像度 2倍
    background: "white",
    type: "image/png",
  });

  if (img) {
    const a = document.createElement("a");
    a.href = (img as HTMLImageElement).src;
    a.download = "diagram.png";
    a.click();
  }
}

// ━━━ Blob として取得（アップロード用） ━━━
function exportBlob(): Promise<Blob | null> {
  return new Promise((resolve) => {
    const imgData = diagram.makeImageData({
      scale: 2,
      background: "white",
      returnType: "blob",
    });
    resolve(imgData as Blob | null);
  });
}
```

## 12. React 統合の完全サンプル

双方向データバインディング + 外部操作の完全例です。

```tsx
// src/components/InteractiveDiagram.tsx
import { useCallback, useRef, useState } from "react";
import * as go from "gojs";
import { ReactDiagram } from "gojs-react";

interface NodeData {
  key: number;
  text: string;
  color: string;
  parent?: number;
}

function initDiagram(): go.Diagram {
  const diagram = new go.Diagram({
    "undoManager.isEnabled": true,
    layout: new go.TreeLayout({ angle: 90, layerSpacing: 40 }),
    model: new go.TreeModel({ nodeKeyProperty: "key", nodeParentKeyProperty: "parent" }),
  });

  diagram.nodeTemplate = new go.Node("Auto")
    .add(
      new go.Shape("RoundedRectangle", {
        fill: "white", stroke: "#333", strokeWidth: 1.5,
      }).bind("fill", "color"),
      new go.TextBlock({
        margin: 10, font: "bold 14px sans-serif", editable: true,
      }).bind(new go.Binding("text").makeTwoWay()) // 双方向バインド
    );

  diagram.linkTemplate = new go.Link({
    routing: go.Routing.Orthogonal, corner: 5,
  }).add(
    new go.Shape({ strokeWidth: 2, stroke: "#555" }),
    new go.Shape({ toArrow: "Standard", fill: "#555", stroke: null })
  );

  return diagram;
}

export default function InteractiveDiagram() {
  const diagramRef = useRef<ReactDiagram>(null);
  const [nodeDataArray, setNodeDataArray] = useState<NodeData[]>([
    { key: 1, text: "ルート",   color: "#FFD700" },
    { key: 2, text: "ノード A", color: "#87CEEB", parent: 1 },
    { key: 3, text: "ノード B", color: "#87CEEB", parent: 1 },
  ]);

  // ダイアグラム上の変更を状態に反映
  const handleModelChange = useCallback((e: go.IncrementalData) => {
    if (e.modifiedNodeData) {
      setNodeDataArray((prev) =>
        prev.map((nd) => {
          const modified = e.modifiedNodeData?.find((m: any) => m.key === nd.key);
          return modified ? { ...nd, ...modified } : nd;
        })
      );
    }
  }, []);

  // 外部からノードを追加
  const addNode = () => {
    const newKey = Math.max(...nodeDataArray.map((n) => n.key)) + 1;
    setNodeDataArray((prev) => [
      ...prev,
      { key: newKey, text: `ノード ${newKey}`, color: "#98FB98", parent: 1 },
    ]);
  };

  // JSON としてエクスポート
  const exportJson = () => {
    const diagram = diagramRef.current?.getDiagram();
    if (diagram) {
      const json = diagram.model.toJson();
      console.log(json);
      alert("コンソールに JSON を出力しました");
    }
  };

  return (
    <div>
      <div style={{ marginBottom: 10, display: "flex", gap: 8 }}>
        <button onClick={addNode}>➕ ノードを追加</button>
        <button onClick={exportJson}>📋 JSON エクスポート</button>
      </div>
      <ReactDiagram
        ref={diagramRef}
        initDiagram={initDiagram}
        divClassName="diagram-component"
        nodeDataArray={nodeDataArray}
        onModelChange={handleModelChange}
      />
      <p>ノード数: {nodeDataArray.length}</p>
    </div>
  );
}
```
