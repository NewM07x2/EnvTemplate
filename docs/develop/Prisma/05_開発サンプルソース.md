## 1. サンプルで使用するスキーマ

以下のスキーマを前提として、各サンプルコードを記載します。

```prisma
// prisma/schema.prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// ユーザー
model User {
  id        Int      @id @default(autoincrement())
  email     String   @unique
  name      String?
  age       Int
  role      Role     @default(USER)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
  deleted   Boolean  @default(false)
  posts     Post[]
  profile   Profile?

  @@index([email])
  @@index([createdAt])
}

// ユーザープロフィール（1対1）
model Profile {
  id     Int     @id @default(autoincrement())
  bio    String?
  avatar String?
  userId Int     @unique
  user   User    @relation(fields: [userId], references: [id], onDelete: Cascade)
}

// 投稿
model Post {
  id        Int        @id @default(autoincrement())
  title     String
  content   String?
  published Boolean    @default(false)
  authorId  Int
  author    User       @relation(fields: [authorId], references: [id], onDelete: Cascade)
  tags      Tag[]
  createdAt DateTime   @default(now())
  updatedAt DateTime   @updatedAt

  @@index([authorId])
  @@index([published, createdAt])
}

// タグ（多対多）
model Tag {
  id    Int    @id @default(autoincrement())
  name  String @unique
  posts Post[]
}

// ロール（列挙型）
enum Role {
  USER
  ADMIN
  EDITOR
}
```

## 2. CRUD 操作（基本編）

### CREATE — データの作成

```typescript
import { prisma } from '@/lib/prisma/client'

// ━━━ 1件作成 ━━━
const user = await prisma.user.create({
  data: {
    email: 'alice@example.com',
    name: 'Alice',
    age: 25,
  },
})
// 戻り値: { id: 1, email: 'alice@example.com', name: 'Alice', age: 25, role: 'USER', ... }

// ━━━ 返却フィールドを限定 ━━━
const userPartial = await prisma.user.create({
  data: {
    email: 'bob@example.com',
    name: 'Bob',
    age: 30,
  },
  select: {
    id: true,
    email: true,
  },
})
// 戻り値: { id: 2, email: 'bob@example.com' }

// ━━━ 複数件一括作成（createMany） ━━━
const result = await prisma.user.createMany({
  data: [
    { email: 'charlie@example.com', name: 'Charlie', age: 28 },
    { email: 'diana@example.com', name: 'Diana', age: 22 },
    { email: 'eve@example.com', name: 'Eve', age: 35 },
  ],
  skipDuplicates: true, // email 重複はスキップ（エラーにならない）
})
// 戻り値: { count: 3 }（作成された件数のみ。個別のレコードは返らない）
```

### READ — データの取得

```typescript
// ━━━ ID で 1 件取得（見つからなければ null） ━━━
const user = await prisma.user.findUnique({
  where: { id: 1 },
})

// ━━━ ID で 1 件取得（見つからなければ例外をスロー） ━━━
const user = await prisma.user.findUniqueOrThrow({
  where: { id: 1 },
})
// NotFoundError: No User found

// ━━━ ユニークフィールドで取得 ━━━
const user = await prisma.user.findUnique({
  where: { email: 'alice@example.com' },
})

// ━━━ 条件で最初の 1 件を取得 ━━━
const user = await prisma.user.findFirst({
  where: {
    age: { gte: 20 },
    deleted: false,
  },
  orderBy: { createdAt: 'desc' },
})

// ━━━ 全件取得 ━━━
const users = await prisma.user.findMany()

// ━━━ 条件付き全件取得 ━━━
const activeUsers = await prisma.user.findMany({
  where: {
    deleted: false,
    role: 'USER',
  },
  orderBy: { createdAt: 'desc' },
})

// ━━━ 件数を取得 ━━━
const count = await prisma.user.count({
  where: { deleted: false },
})
```

### UPDATE — データの更新

```typescript
// ━━━ ID で 1 件更新 ━━━
const updated = await prisma.user.update({
  where: { id: 1 },
  data: { name: 'Alice Updated', age: 26 },
})

// ━━━ 条件に一致する全件を更新 ━━━
const result = await prisma.user.updateMany({
  where: {
    role: 'USER',
    age: { lt: 18 },
  },
  data: { role: 'USER' }, // ロールを強制的に USER に
})
// 戻り値: { count: 5 }（更新された件数）

// ━━━ 数値のインクリメント ━━━
const updated = await prisma.user.update({
  where: { id: 1 },
  data: {
    age: { increment: 1 }, // age を +1
  },
})
// 他: decrement（減算）/ multiply（乗算）/ divide（除算）
```

### DELETE — データの削除

```typescript
// ━━━ ID で 1 件削除 ━━━
const deleted = await prisma.user.delete({
  where: { id: 1 },
})

// ━━━ 条件に一致する全件を削除 ━━━
const result = await prisma.user.deleteMany({
  where: {
    deleted: true, // 論理削除済みのレコードを物理削除
  },
})
// 戻り値: { count: 3 }

// ━━━ 全件削除（テストのクリーンアップ等） ━━━
await prisma.user.deleteMany()

// ━━━ 論理削除（ソフトデリート）━━━
const softDeleted = await prisma.user.update({
  where: { id: 1 },
  data: { deleted: true },
})
```

## 3. リレーション操作

### 関連データを含めて取得（include）

```typescript
// ━━━ User と Post を一緒に取得 ━━━
const userWithPosts = await prisma.user.findUnique({
  where: { id: 1 },
  include: {
    posts: true,     // 全投稿を含める
    profile: true,   // プロフィールを含める
  },
})
// userWithPosts.posts → Post[]
// userWithPosts.profile → Profile | null

// ━━━ ネストした include ━━━
const userWithPostsAndTags = await prisma.user.findUnique({
  where: { id: 1 },
  include: {
    posts: {
      include: {
        tags: true,  // 各投稿のタグも含める
      },
      where: {
        published: true, // 公開済みの投稿のみ
      },
      orderBy: {
        createdAt: 'desc',
      },
    },
  },
})

// ━━━ select と include の組み合わせ ━━━
const userSummary = await prisma.user.findUnique({
  where: { id: 1 },
  select: {
    id: true,
    name: true,
    posts: {
      select: {
        id: true,
        title: true,
      },
      take: 5, // 最新 5 件のみ
    },
    _count: {
      select: { posts: true }, // 投稿の総数
    },
  },
})
// { id: 1, name: 'Alice', posts: [...], _count: { posts: 10 } }
```

### 関連データを同時に作成（Nested Create）

```typescript
// ━━━ User + Post を同時に作成 ━━━
const user = await prisma.user.create({
  data: {
    email: 'alice@example.com',
    name: 'Alice',
    age: 25,
    posts: {
      create: [
        { title: '最初の投稿', content: 'Hello World!', published: true },
        { title: '2 番目の投稿', content: 'Prisma は便利' },
      ],
    },
    profile: {
      create: {
        bio: 'フロントエンドエンジニア',
        avatar: 'https://example.com/avatar.png',
      },
    },
  },
  include: {
    posts: true,
    profile: true,
  },
})

// ━━━ 既存の User に Post を追加 ━━━
const post = await prisma.post.create({
  data: {
    title: '新しい投稿',
    content: '追加の投稿です',
    author: {
      connect: { id: 1 }, // 既存の User ID = 1 に紐付け
    },
  },
})

// ━━━ Post + Tag を同時に作成（多対多） ━━━
const post = await prisma.post.create({
  data: {
    title: 'TypeScript 入門',
    content: 'TypeScript について...',
    authorId: 1,
    tags: {
      // 存在すれば接続、なければ作成
      connectOrCreate: [
        {
          where: { name: 'TypeScript' },
          create: { name: 'TypeScript' },
        },
        {
          where: { name: 'Programming' },
          create: { name: 'Programming' },
        },
      ],
    },
  },
  include: { tags: true },
})
```

### 関連データの更新・切断

```typescript
// ━━━ 関連を切断（disconnect） ━━━
// 多対多: Post からタグを外す
const post = await prisma.post.update({
  where: { id: 1 },
  data: {
    tags: {
      disconnect: [{ id: 1 }], // Tag ID = 1 を外す
    },
  },
})

// ━━━ 関連を置き換え（set） ━━━
// 多対多: Post のタグを完全に入れ替え
const post = await prisma.post.update({
  where: { id: 1 },
  data: {
    tags: {
      set: [{ id: 2 }, { id: 3 }], // Tag ID 2,3 のみに設定
    },
  },
})
```

## 4. フィルタ・検索

### 基本的なフィルタ

```typescript
// ━━━ 等値・不等値 ━━━
const users = await prisma.user.findMany({
  where: {
    role: 'ADMIN',           // 等値
    age: { not: 0 },         // 不等値
  },
})

// ━━━ 数値の範囲 ━━━
const users = await prisma.user.findMany({
  where: {
    age: {
      gte: 20,  // >= 20
      lte: 30,  // <= 30
    },
  },
})
// gt: >  /  gte: >=  /  lt: <  /  lte: <=

// ━━━ 文字列の部分一致 ━━━
const users = await prisma.user.findMany({
  where: {
    name: { contains: 'ali' },      // 部分一致（'ali' を含む）
    // name: { startsWith: 'A' },   // 前方一致
    // name: { endsWith: 'ce' },    // 後方一致
  },
})

// ━━━ 大文字小文字を無視した検索 ━━━
const users = await prisma.user.findMany({
  where: {
    name: {
      contains: 'alice',
      mode: 'insensitive', // 大文字小文字を区別しない
    },
  },
})

// ━━━ IN（いずれかに一致） ━━━
const users = await prisma.user.findMany({
  where: {
    id: { in: [1, 2, 3] },
    // id: { notIn: [4, 5] },  // いずれにも一致しない
  },
})

// ━━━ NULL チェック ━━━
const usersWithName = await prisma.user.findMany({
  where: {
    name: { not: null }, // name が NULL でない
    // name: null,       // name が NULL
  },
})
```

### 複合条件（AND / OR / NOT）

```typescript
// ━━━ AND（すべて満たす）— デフォルト ━━━
const users = await prisma.user.findMany({
  where: {
    // where 直下の条件は暗黙の AND
    deleted: false,
    age: { gte: 20 },
    role: 'USER',
  },
})

// ━━━ OR（いずれかを満たす） ━━━
const users = await prisma.user.findMany({
  where: {
    OR: [
      { email: { contains: '@example.com' } },
      { role: 'ADMIN' },
    ],
  },
})

// ━━━ NOT（否定） ━━━
const users = await prisma.user.findMany({
  where: {
    NOT: {
      role: 'ADMIN',
    },
  },
})

// ━━━ 複雑な組み合わせ ━━━
const users = await prisma.user.findMany({
  where: {
    AND: [
      { deleted: false },
      {
        OR: [
          { age: { gte: 18, lte: 30 } },
          { role: 'ADMIN' },
        ],
      },
    ],
  },
})
```

### リレーションを使ったフィルタ

```typescript
// ━━━ 「投稿がある」ユーザーを取得 ━━━
const usersWithPosts = await prisma.user.findMany({

// ━━━ 「全ての投稿が公開済み」のユーザー ━━━
const users = await prisma.user.findMany({

    posts: {

    },
  },
})
```

## 5. ソート・ページネーション

### ソート

```typescript
// ━━━ 単一カラムでソート ━━━
const users = await prisma.user.findMany({
  orderBy: { createdAt: 'desc' }, // 新しい順
})

// ━━━ 複数カラムでソート ━━━
const users = await prisma.user.findMany({
  orderBy: [
    { role: 'asc' },        // まずロール順
    { createdAt: 'desc' },  // 同じロール内で新しい順
  ],
})

// ━━━ NULL の並び順 ━━━
const users = await prisma.user.findMany({
  orderBy: {
    name: { sort: 'asc', nulls: 'last' }, // NULL を最後に
  },
})

// ━━━ リレーションの件数でソート ━━━
const users = await prisma.user.findMany({
  orderBy: {
    posts: { _count: 'desc' }, // 投稿数が多い順
  },
})
```

### オフセットベースのページネーション

```typescript
// ━━━ ページネーション関数 ━━━
async function getUsers(page: number, pageSize: number = 10) {
  const [users, total] = await Promise.all([
    prisma.user.findMany({
      where: { deleted: false },
      skip: (page - 1) * pageSize,
      take: pageSize,
      orderBy: { createdAt: 'desc' },
    }),
    prisma.user.count({
      where: { deleted: false },
    }),
  ])

  return {
    data: users,
    pagination: {
      currentPage: page,
      pageSize,
      totalCount: total,
      totalPages: Math.ceil(total / pageSize),
      hasNext: page * pageSize < total,
      hasPrev: page > 1,
    },
  }
}

// 使用例
const result = await getUsers(1, 10)
// {
//   data: [...],
//   pagination: {
//     currentPage: 1, pageSize: 10, totalCount: 100,
//     totalPages: 10, hasNext: true, hasPrev: false
//   }
// }
```

### カーソルベースのページネーション

```typescript
// ━━━ 大量データ向け（推奨） ━━━
async function getUsersCursor(cursor?: number, pageSize: number = 10) {
  const users = await prisma.user.findMany({
    where: { deleted: false },
    take: pageSize + 1, // 次ページがあるか判定するため +1
    ...(cursor && {
      cursor: { id: cursor },
      skip: 1, // カーソル自身をスキップ
    }),
    orderBy: { id: 'asc' },
  })

  const hasNext = users.length > pageSize
  if (hasNext) users.pop() // 余分な1件を削除

  return {
    data: users,
    nextCursor: hasNext ? users[users.length - 1].id : null,
  }
}

// 使用例
const page1 = await getUsersCursor()          // 最初のページ
const page2 = await getUsersCursor(page1.nextCursor!) // 次のページ
```

## 6. 集計・グルーピング

```typescript
// ━━━ 基本的な集計 ━━━
const stats = await prisma.user.aggregate({
  _count: { _all: true },   // 全件数
  _avg: { age: true },       // 年齢の平均
  _max: { age: true },       // 年齢の最大値
  _min: { age: true },       // 年齢の最小値
  _sum: { age: true },       // 年齢の合計
  where: { deleted: false },
})
// { _count: { _all: 50 }, _avg: { age: 28.5 }, _max: { age: 65 }, ... }

// ━━━ グルーピング ━━━
const usersByRole = await prisma.user.groupBy({
  by: ['role'],
  _count: { _all: true },
  _avg: { age: true },
  where: { deleted: false },
  orderBy: { _count: { role: 'desc' } },
})
// [
//   { role: 'USER',   _count: { _all: 40 }, _avg: { age: 27 } },
//   { role: 'ADMIN',  _count: { _all: 5 },  _avg: { age: 35 } },
//   { role: 'EDITOR', _count: { _all: 5 },  _avg: { age: 30 } },
// ]

// ━━━ HAVING（グループに条件をつける） ━━━
const popularRoles = await prisma.user.groupBy({
  by: ['role'],
  _count: { _all: true },
  having: {
    _all: { _count: { gte: 5 } }, // 5人以上のロールのみ
  },
})
```

## 7. トランザクション

### バッチトランザクション（配列形式）

```typescript
// 複数の操作を同時に実行（全て成功 or 全てロールバック）
const [user, post, tagCount] = await prisma.$transaction([
  prisma.user.create({
    data: { email: 'alice@example.com', name: 'Alice', age: 25 },
  }),
  prisma.post.create({
    data: { title: '投稿', authorId: 1 },
  }),
  prisma.tag.count(),
])
```

### インタラクティブトランザクション（コールバック形式）

```typescript
// ★ 推奨：エラー処理が明確で、前の操作の結果を後続で使える
const result = await prisma.$transaction(async (tx) => {
  // 1. ユーザーを作成
  const user = await tx.user.create({
    data: { email: 'alice@example.com', name: 'Alice', age: 25 },
  })

  // 2. 作成したユーザーの ID を使って投稿を作成
  const post = await tx.post.create({
    data: {
      title: 'Alice の最初の投稿',
      authorId: user.id, // ← 上で作成した user の ID を使用
    },
  })

  // 3. プロフィールを作成
  const profile = await tx.profile.create({
    data: {
      bio: 'はじめまして',
      userId: user.id,
    },
  })

  return { user, post, profile }
  // ↑ ここまで全て成功したらコミット
  // どこかでエラーが起きたら全てロールバック
})
```

### トランザクションのオプション

```typescript
await prisma.$transaction(
  async (tx) => {
    // 重い処理...
  },
  {
    maxWait: 5000,     // トランザクション取得の最大待機時間（ms）
    timeout: 10000,    // トランザクションの最大実行時間（ms）
    isolationLevel: 'Serializable', // 分離レベル
  }
)
// 分離レベル: ReadUncommitted / ReadCommitted / RepeatableRead / Serializable
```

### 実践例: 銀行の送金処理

```typescript
async function transfer(fromId: number, toId: number, amount: number) {
  return prisma.$transaction(async (tx) => {
    // 1. 送金元の残高を確認
    const from = await tx.account.findUniqueOrThrow({
      where: { id: fromId },
    })

    if (from.balance < amount) {
      throw new Error('残高不足です')
    }

    // 2. 送金元から引き落とし
    await tx.account.update({
      where: { id: fromId },
      data: { balance: { decrement: amount } },
    })

    // 3. 送金先に入金
    await tx.account.update({
      where: { id: toId },
      data: { balance: { increment: amount } },
    })

    // 4. 取引履歴を記録
    await tx.transaction.create({
      data: {
        fromAccountId: fromId,
        toAccountId: toId,
        amount,
        type: 'TRANSFER',
      },
    })
  })
}
```

## 9. 生 SQL（$queryRaw / $executeRaw）

Prisma API では実現できない複雑なクエリが必要な場合に使用します。

```typescript
import { Prisma } from '@prisma/client'

// ━━━ SELECT（型付き） ━━━
const users = await prisma.$queryRaw<
  { id: number; name: string; post_count: bigint }[]
>`
  SELECT u.id, u.name, COUNT(p.id) as post_count
  FROM "User" u
  LEFT JOIN "Post" p ON p."authorId" = u.id
  WHERE u.deleted = false
  GROUP BY u.id
  HAVING COUNT(p.id) > 0
  ORDER BY post_count DESC
  LIMIT 10
`

// ━━━ パラメータ付き（SQLインジェクション防止） ━━━
const minAge = 20
const role = 'USER'
const users = await prisma.$queryRaw`
  SELECT * FROM "User"
  WHERE age >= ${minAge}
  AND role = ${role}::text
  AND deleted = false
`
// テンプレートリテラルで渡した変数は自動的にパラメータバインドされる（安全）

// ━━━ INSERT / UPDATE / DELETE（$executeRaw） ━━━
const affectedRows = await prisma.$executeRaw`
  UPDATE "User"
  SET deleted = true
  WHERE "createdAt" < NOW() - INTERVAL '1 year'
  AND deleted = false
`
// 戻り値: 影響を受けた行数（number）

// ━━━ 動的なクエリ（Prisma.sql） ━━━
const columns = Prisma.sql`id, name, email`
const table = Prisma.sql`"User"`
const condition = Prisma.sql`deleted = false`

const users = await prisma.$queryRaw`
  SELECT ${columns} FROM ${table} WHERE ${condition}
`
```

> **⚠️ 注意:** `$queryRaw` でテンプレートリテラルを使えば安全ですが、文字列連結 (`$queryRawUnsafe`) は**SQLインジェクションのリスク**があります。可能な限り避けてください。

```typescript
// src/app/api/users/route.ts
import { prisma } from '@/lib/prisma/client'
import { NextRequest, NextResponse } from 'next/server'

// GET /api/users — ユーザー一覧取得
export async function GET(request: NextRequest) {
  const searchParams = request.nextUrl.searchParams
  const page = Number(searchParams.get('page') ?? '1')
  const pageSize = Number(searchParams.get('pageSize') ?? '10')
  const search = searchParams.get('search') ?? ''

  try {
    const where = {
      deleted: false,
      ...(search && {
        OR: [
          { name: { contains: search, mode: 'insensitive' as const } },
          { email: { contains: search, mode: 'insensitive' as const } },
        ],
      }),
    }

    const [users, total] = await Promise.all([
      prisma.user.findMany({
        where,
        skip: (page - 1) * pageSize,
        take: pageSize,
        orderBy: { createdAt: 'desc' },
        select: {
          id: true,
          email: true,
          name: true,
          age: true,
          role: true,
          createdAt: true,
          _count: { select: { posts: true } },
        },
      }),
      prisma.user.count({ where }),
    ])

    return NextResponse.json({
      data: users,
      pagination: {
        currentPage: page,
        pageSize,
        totalCount: total,
        totalPages: Math.ceil(total / pageSize),
      },
    })
  } catch (error) {
    console.error('ユーザー取得エラー:', error)
    return NextResponse.json(
      { error: 'ユーザーの取得に失敗しました' },
      { status: 500 }
    )
  }
}

// POST /api/users — ユーザー作成
export async function POST(request: NextRequest) {
  try {
    const body = await request.json()
    const { email, name, age } = body

    // バリデーション
    if (!email || !age) {
      return NextResponse.json(
        { error: 'email と age は必須です' },
        { status: 400 }
      )
    }

    const user = await prisma.user.create({
      data: { email, name, age },
    })

    return NextResponse.json(user, { status: 201 })
  } catch (error: any) {
    // ユニーク制約違反
    if (error.code === 'P2002') {
      return NextResponse.json(
        { error: 'このメールアドレスは既に使用されています' },
        { status: 409 }
      )
    }
    console.error('ユーザー作成エラー:', error)
    return NextResponse.json(
      { error: 'ユーザーの作成に失敗しました' },
      { status: 500 }
    )
  }
}
```

```typescript
  request: NextRequest,

) {
  const id = Number(params.id)

  const user = await prisma.user.findUnique({
    where: { id },

        where: { published: true },
        orderBy: { createdAt: 'desc' },

    return NextResponse.json(
      { error: 'ユーザーが見つかりません' },
      { status: 404 }
    )
  }

  return NextResponse.json(user)
}

export async function PATCH(
  request: NextRequest,

) {

        { status: 404 }
```

### エラーコードの一覧（よく遭遇するもの）

| コード | 意味 | よくある原因 |
|-------|------|------------|
| `P2002` | ユニーク制約違反 | 重複する email 等で create/update |
| `P2003` | 外部キー制約違反 | 存在しない authorId で create |
| `P2025` | レコード未検出 | update / delete 対象が存在しない |
| `P2014` | リレーション違反 | 必須リレーションの削除 |
| `P2021` | テーブル未存在 | マイグレーション未実行 |

```typescript
// エラーハンドリングの例
import { Prisma } from '@prisma/client'

try {
  await prisma.user.create({ data: { email, name, age } })
} catch (error) {
  if (error instanceof Prisma.PrismaClientKnownRequestError) {
    switch (error.code) {
      case 'P2002':
        throw new Error('このメールアドレスは既に使用されています')
      case 'P2003':
        throw new Error('参照先のデータが存在しません')

        throw new Error('データベースエラーが発生しました')
```
